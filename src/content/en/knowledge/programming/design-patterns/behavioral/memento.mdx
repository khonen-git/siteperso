{`class EditorMemento:
    def __init__(self, text):
        self.text = text

class Editor:
    def __init__(self):
        self.text = ""

    def type(self, chars):
        self.text += chars

    def save(self):
        return EditorMemento(self.text)

    def restore(self, memento):
        self.text = memento.text

editor = Editor()
editor.type("Hello")
saved = editor.save()
editor.type(" World")
print(editor.text)
editor.restore(saved)
