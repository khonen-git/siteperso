## Python example

<CodeBlock language="python">
{`class LockedState:
    def insert_coin(self, machine):
        print("Coin accepted")
        machine.state = UnlockedState()

    def eject(self, machine):
        print("Insert coin first")

class UnlockedState:
    def insert_coin(self, machine):
        print("Already unlocked")

    def eject(self, machine):
        print("Door open")
        machine.state = LockedState()

class Turnstile:
    def __init__(self):
        self.state = LockedState()

    def insert_coin(self):
        self.state.insert_coin(self)

    def push(self):
        self.state.eject(self)

t = Turnstile()
t.insert_coin()
t.push()
