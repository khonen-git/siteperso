{`class RealImage:
    def __init__(self, path):
        self.path = path
        print(f"Loading {path}")

class ImageProxy:
    def __init__(self, path):
        self.path = path
        self._real = None

    def display(self):
        if self._real is None:
            self._real = RealImage(self.path)
        self._real.display()
