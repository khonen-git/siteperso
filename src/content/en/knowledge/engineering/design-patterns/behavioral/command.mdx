## Python example

<CodeBlock language="python">
{`class Light:
    def __init__(self):
        self.on = False

    def toggle(self):
        self.on = not self.on
        return "ON" if self.on else "OFF"

class Command:
    def __init__(self, light):
        self.light = light

    def execute(self):
        return self.light.toggle()

class Remote:
    def __init__(self):
        self.history = []

    def press(self, cmd):
        state = cmd.execute()
        self.history.append(cmd)
        print(f"Light is {state}")

    def undo(self):
        if self.history:
            state = self.history.pop().execute()
            print(f"Undo -> {state}")

remote = Remote()
remote.press(Command(Light()))
remote.undo()`}
